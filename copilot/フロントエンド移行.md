```
    ├── generatedApi.ts      # OpenAPI自動生成
    └── enhancedApi.ts       # キャッシュ無効化等
```

### コマンド
```bash
# ビルド
cd frontend && npm run build

# SCSS型生成
npm run scss:types

# API再生成（バックエンド変更後）
npm run generate-api

# テスト
npm run test:run
```
